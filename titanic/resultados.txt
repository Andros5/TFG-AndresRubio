def calcular_base_stem(self):
        """Calcula la base Stem de un contexto formal como lista de tuplas de conjuntos de atributos."""
        pseudointenciones = []  # Lista de pseudointenciones como tuplas de atributos
        atributos = self.properties  # Atributos del contexto como tupla
        for i in range(len(atributos) + 1):
            pseudointencionesAux = []
            for comb in combinations(atributos, i):  # Recorro todas las combinaciones de i atributos
                pseudoIntSet = set(comb)  # Creo un conjunto con la combinación actual
                if pseudoIntSet != set(self.intension(self.extension(comb))):  # P=/P''
                    valido = True
                    for subIntSet in pseudointenciones:  # Recorro las pseudointenciones ya calculadas
                        if subIntSet.issubset(pseudoIntSet):  # Verifico si Q'' está contenido en P
                            subIntppSet = set(self.intension(self.extension(subIntSet)))
                            if not subIntppSet.issubset(pseudoIntSet):
                                valido = False
                                break
                    if valido:
                        pseudointencionesAux.append(pseudoIntSet)
            pseudointenciones.extend(pseudointencionesAux)
        self.base_stem = [(p, set(self.intension(self.extension(p))) - p) for p in pseudointenciones]
        return self.base_stem

def calcular_base_luxenburger(self):
        """Calcula la base de Luxenburger de un contexto formal como lista de tuplas de conjuntos de atributos."""
        base_luxenburger = []
        visitados = set()

        def aux_base_luxenburger(concepto_actual):
            """Auxiliar para recorrer cada nodo del retículo del contexto formal."""
            if concepto_actual in visitados: # Si el concepto actual ya fue visitado, no hacer nada
                return
            visitados.add(concepto_actual)
            for vecino in self.neighbors(concepto_actual[0]): # Recorrer los vecinos del concepto actual
                intencionVecino = set(vecino[1])
                intencionActual = set(concepto_actual[1])
                implicacion = (intencionVecino, intencionActual - intencionVecino)
                conf_Implicacion = self.confianza_implicacion(implicacion)
                sop_Implicacion = self.soporte_implicacion(implicacion)
                if sop_Implicacion >= self.soporte and conf_Implicacion >= self.confianza: # Si la implicación es frecuente, agregarla a la base de Luxenburger
                    base_luxenburger.append((implicacion, conf_Implicacion, sop_Implicacion))
                aux_base_luxenburger(vecino)

        atributos = self.properties
        aux_base_luxenburger((self.extension(atributos), atributos))
        self.base_luxenburger = base_luxenburger
        return self.base_luxenburger
